def display_tweets hash_tag, tweets
  begin
    puts "\n\n=======Displaying Recent 20 Tweets That Include the Hash Tag: ##{hash_tag}========\n\n"
    tweets.take(20).each_with_index do |tweet, index|
      puts "\n#{index + 1}. \n User: #{tweet.user.screen_name}"
      puts "\n Tweeted On: #{tweet.created_at.strftime('%d%b, %Y at %I:%M%p')}"
      puts "\n\n Tweet: \n  #{tweet.text}\n\n"
      puts "---------------------------------------------------------------------------"
    end
    puts "\n\n==========================Tweets End Here====================================\n\n\n\n"
  rescue Exception => e
    log.error "There was an error while displaying tweets: #{e.message}"
    raise
  end
end

def get_hash_tag
  begin
    hash_tag = ""
    while hash_tag.strip.length == 0 do
      puts "\n\nPlease Enter Hash Tag"
      STDOUT.flush
      hash_tag = gets.chomp
      puts "You Have Entered: ##{hash_tag}" unless hash_tag.strip.length == 0
    end
    get_recent_tweets( hash_tag )
  rescue Exception => e
    log.error "There was an error while accepting Hash Tag as an input: #{e.message}"
    raise
  end
end

def get_recent_tweets hash_tag
  begin
    get_twitter
    @tweets = @client.search( "#"+hash_tag, count: 20, result_type: "recent" )
    display_tweets( hash_tag, @tweets )
  rescue Exception => e
    log.error "There was an error while retrieving tweets: #{e.message}"
    raise
  end
end
